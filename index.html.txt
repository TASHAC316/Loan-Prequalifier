  <form id="loanForm">
    <label>Annual Income ($):</label>
    <input type="number" id="income" required />

    <label>Credit Score:</label>
    <input type="number" id="creditScore" required />

    <label>Monthly Debt Payments ($):</label>
    <input type="number" id="debt" required />

    <button type="submit">Check Qualification</button>
  </form>

  <div id="results"></div>

  <script>
    document.getElementById("loanForm").addEventListener("submit", function(event) {
      event.preventDefault();

      const income = parseFloat(document.getElementById("income").value);
      const creditScore = parseInt(document.getElementById("creditScore").value);
      const debt = parseFloat(document.getElementById("debt").value);
      const monthlyIncome = income / 12;
      const dti = (debt / monthlyIncome) * 100;

      const banks = [
        {
          name: "Bank A",
          minIncome: 50000,
          minCreditScore: 680,
          maxDTI: 40,
          contactLink: "https://calendly.com/bank-a"
        },
        {
          name: "Bank B",
          minIncome: 40000,
          minCreditScore: 640,
          maxDTI: 50,
          contactLink: "mailto:bankb@loans.com"
        }
      ];

      let resultsHTML = "<h2>Qualification Results:</h2>";

      banks.forEach(bank => {
        const qualifies =
          income >= bank.minIncome &&
          creditScore >= bank.minCreditScore &&
          dti <= bank.maxDTI;

        resultsHTML += `
          <div class="bank">
            <strong>${bank.name}</strong> - 
            ${qualifies ? `<span class="qualified">You Qualify ✅</span><br><a href="${bank.contactLink}" target="_blank">Contact Banker</a>` : `<span class="not-qualified">You Do Not Qualify ❌</span>`}
          </div>
        `;
      });

      document.getElementById("results").innerHTML = resultsHTML;
    });
  </script>
